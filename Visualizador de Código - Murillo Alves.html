Este projeto utiliza um sensor AHT10 para medir temperatura e umidade, enviando os dados para o app Blynk via ESP32. Permite monitoramento remoto em tempo real, facilitando o desenvolvimento em sistemas embarcados.
Zoom do código: 

#define BLYNK_TEMPLATE_ID "TMPL2IJ0DFQyk"
#define BLYNK_TEMPLATE_NAME "AHT10"
#define BLYNK_AUTH_TOKEN "4e5AIKGIx146aUn5QJH4wdDOVzCdEw5j"

#define BLYNK_PRINT Serial
#include <WiFi.h>
#include <WiFiClient.h>
#include <BlynkSimpleEsp32.h>
#include <Adafruit_AHT10.h>

Adafruit_AHT10 aht;
Adafruit_Sensor *aht_humidity, *aht_temp;

char ssid[] = "senai109";
char pass[] = "senai109";

BlynkTimer timer;

float umidade;
float temperatura;

void EnviaValor() {
  Blynk.virtualWrite(V0, umidade);
  Blynk.virtualWrite(V1, temperatura);
}

void setup() {
  Serial.begin(115200);
  Blynk.begin(BLYNK_AUTH_TOKEN, ssid, pass);
  timer.setInterval(1000L, EnviaValor);
  while (!Serial) delay(10);

  Serial.println("Adafruit AHT10 teste!");

  if (!aht.begin()) {
    Serial.println("Erro ao tentar encontrar o sensor AHT10");
    while (1) delay(10);
  }

  Serial.println("AHT10 Encontrado!");
  aht_temp = aht.getTemperatureSensor();
  aht_temp->printSensorDetails();

  aht_humidity = aht.getHumiditySensor();
  aht_humidity->printSensorDetails();
}

void loop() {
  sensors_event_t humidity;
  sensors_event_t temp;

  aht_humidity->getEvent(&humidity);
  aht_temp->getEvent(&temp);

  umidade = humidity.relative_humidity;
  temperatura = temp.temperature;

  Blynk.run();
  timer.run();
}
